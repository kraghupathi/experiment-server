#+BEGIN_SRC
vzctl create 31 --ostemplate ubuntu-14.04-x86_64 --ipadd 10.4.12.31 --hostname experiment-server-labs
#+END_SRC

   - Then start the container as:
#+BEGIN_SRC
vzctl start 31
#+END_SRC
   - Enter into the container as :
#+BEGIN_SRC
vzctl enter 31
#+END_SRC
